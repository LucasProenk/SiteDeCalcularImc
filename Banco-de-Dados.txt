CREATE DATABASE calculadorimc;
USE calculadorimc;

CREATE TABLE aluno (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nome VARCHAR(100),
  email VARCHAR(100) UNIQUE,
  senha VARCHAR(255)
);

CREATE TABLE imc (
  id INT AUTO_INCREMENT PRIMARY KEY,
  aluno_id INT,
  peso DECIMAL(5,2),
  altura DECIMAL(4,2),
  imc DECIMAL(5,2),
  FOREIGN KEY (aluno_id) REFERENCES aluno(id)
);

--Views
CREATE VIEW registro AS
SELECT 
  a.id,
  a.nome,
  a.email,
  i.peso,
  i.altura,
  i.imc
FROM aluno a
JOIN imc i ON i.aluno_id = a.id;

--Trigger
DELIMITER //
CREATE TRIGGER calcular_imc
BEFORE INSERT ON imc
FOR EACH ROW
BEGIN
  SET NEW.imc = NEW.peso / (NEW.altura * NEW.altura);
END //
DELIMITER ;
